270template <class T> bool Vertex<T>::removeEdge(T in) {
271 bool removedEdge = false;
272 auto it = adj.begin();
273 while (it != adj.end()) {
274 Edge<T> *edge = *it;
275 Vertex<T> *dest = edge->getDest();
276 if (dest->getInfo() == in) {
277 it = adj.erase(it);
278 deleteEdge(edge);
279 removedEdge = true; // allows for multiple edges to connect the same pair
280 // of vertices (multigraph)
281 } else {
282 it++;
283 }
284 }
285 return removedEdge;
286}

288template <class T> void Vertex<T>::removeOutgoingEdges() {
289 auto it = adj.begin();
290 while (it != adj.end()) {
291 Edge<T> *edge = *it;
292 it = adj.erase(it);
293 deleteEdge(edge);
294 }
295}

348 while (it != dest->incoming.end()) {

448template <class T> bool Graph<T>::removeVertex(const T &in) {
449 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
450 if ((*it)->getInfo() == in) {
451 auto v = *it;
452 v->removeOutgoingEdges();
453 for (auto u : vertexSet) {
454 u->removeEdge(v->getInfo());
455 }
456 vertexSet.erase(it);
457 delete v;
458 return true;
459 }
460 }
461 return false;
462}

482template <class T> Edge<T> *Graph<T>::findEdge(int sourceId, int destId) {
483 auto sourceVertex = findVertexById(sourceId);
484 if (!sourceVertex)
485 return nullptr;
486
487 for (auto &edge : sourceVertex->getAdj()) {
488 if (edge->getDest()->getInfo().id == destId) {
489 return edge;
490 }
491 }
492 return nullptr;
493}

495template <class T>
496bool Graph<T>::addBidirectionalEdge(const T &sourc, const T &dest, double dw,
497 double ww) {
498 auto v1 = findVertex(sourc);
499 auto v2 = findVertex(dest);
500 if (v1 == nullptr || v2 == nullptr)
501 return false;
502 auto e1 = v1->addEdge(v2, dw, ww);
503 auto e2 = v2->addEdge(v1, dw, ww);
504 e1->setReverse(e2);
505 e2->setReverse(e1);
506 return true;
507}
