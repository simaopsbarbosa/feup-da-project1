177template <class T> bool Vertex<T>::removeEdge(T in) {
178 bool removedEdge = false;
179 auto it = adj.begin();
180 while (it != adj.end()) {
181 Edge<T> *edge = *it;
182 Vertex<T> *dest = edge->getDest();
183 if (dest->getInfo() == in) {
184 it = adj.erase(it);
185 deleteEdge(edge);
186 removedEdge = true; // allows for multiple edges to connect the same pair
187 // of vertices (multigraph)
188 } else {
189 it++;
190 }
191 }
192 return removedEdge;
193}

367template <class T> void Vertex<T>::deleteEdge(Edge<T> *edge) {
368 Vertex<T> *dest = edge->getDest();
369 auto it = dest->incoming.begin();
370 while (it != dest->incoming.end()) {
371 if ((*it)->getOrig()->getInfo() == info) {

575template <class T> bool Graph<T>::removeVertex(const T &in) {
576 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
577 if ((*it)->getInfo() == in) {
578 auto v = *it;
579 v->removeOutgoingEdges();
580 for (auto u : vertexSet) {
581 u->removeEdge(v->getInfo());
582 }
583 vertexSet.erase(it);
584 delete v;
585 return true;
586 }
587 }
588 return false;
589}

635template <class T> Edge<T> *Graph<T>::findEdge(int sourceId, int destId) {
636 auto sourceVertex = findVertexById(sourceId);
637 if (!sourceVertex)
638 return nullptr;
639
640 for (auto &edge : sourceVertex->getAdj()) {
641 if (edge->getDest()->getInfo().id == destId) {
642 return edge;
643 }
644 }
645 return nullptr;
646}

658template <class T>
659bool Graph<T>::addBidirectionalEdge(const T &sourc, const T &dest, double dw,
660 double ww) {
661 auto v1 = findVertex(sourc);
662 auto v2 = findVertex(dest);
663 if (v1 == nullptr || v2 == nullptr)
664 return false;
665 auto e1 = v1->addEdge(v2, dw, ww);
666 auto e2 = v2->addEdge(v1, dw, ww);
667 e1->setReverse(e2);
668 e2->setReverse(e1);
669 return true;
670}
