177 * Auxiliary function to add an outgoing edge to a vertex (this),

180template <class T> Edge<T> *Vertex<T>::addEdge(Vertex<T> *d, double dw, double ww) {
181 auto newEdge = new Edge<T>(this, d, dw, ww);
182 adj.push_back(newEdge);
183 d->incoming.push_back(newEdge);
184 return newEdge;
185}
186
187/*
188 * Auxiliary function to remove an outgoing edge (with a given destination (d))
189 * from a vertex (this).
190 * Returns true if successful, and false if such edge does not exist.
191 */
192template <class T> bool Vertex<T>::removeEdge(T in) {
193 bool removedEdge = false;
194 auto it = adj.begin();
195 while (it != adj.end()) {
196 Edge<T> *edge = *it;
197 Vertex<T> *dest = edge->getDest();
198 if (dest->getInfo() == in) {
199 it = adj.erase(it);
200 deleteEdge(edge);
201 removedEdge = true; // allows for multiple edges to connect the same pair
202 // of vertices (multigraph)
203 } else {
204 it++;
205 }
206 }
207 return removedEdge;
208}

306template <class T> void Vertex<T>::deleteEdge(Edge<T> *edge) {
307 Vertex<T> *dest = edge->getDest();
308 // Remove the corresponding edge from the incoming list
309 auto it = dest->incoming.begin();
310 while (it != dest->incoming.end()) {
311 if ((*it)->getOrig()->getInfo() == info) {
312 it = dest->incoming.erase(it);
313 } else {
314 it++;
315 }
316 }
317 delete edge;
318}

435/*
436 * Removes a vertex with a given content (in) from a graph (this), and
437 * all outgoing and incoming edges.
438 * Returns true if successful, and false if such vertex does not exist.
439 */
440template <class T> bool Graph<T>::removeVertex(const T &in) {
441 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
442 if ((*it)->getInfo() == in) {
443 auto v = *it;
444 v->removeOutgoingEdges();
445 for (auto u : vertexSet) {
446 u->removeEdge(v->getInfo());
447 }
448 vertexSet.erase(it);
449 delete v;
450 return true;
451 }
452 }
453 return false;
454}

484template <class T> Edge<T> *Graph<T>::findEdge(int sourceId, int destId) {
485 auto sourceVertex = findVertexById(sourceId);
486 if (!sourceVertex)
487 return nullptr;
488
489 for (auto &edge : sourceVertex->getAdj()) {
490 if (edge->getDest()->getInfo().id == destId) {
491 return edge;
492 }
493 }
494 return nullptr;
495}

497template <class T> bool Graph<T>::addBidirectionalEdge(const T &sourc, const T &dest, double dw, double ww) {
498 auto v1 = findVertex(sourc);
499 auto v2 = findVertex(dest);
500 if (v1 == nullptr || v2 == nullptr)
501 return false;
502 auto e1 = v1->addEdge(v2, dw, ww);
503 auto e2 = v2->addEdge(v1, dw, ww);
504 e1->setReverse(e2);
505 e2->setReverse(e1);
506 return true;
507}
